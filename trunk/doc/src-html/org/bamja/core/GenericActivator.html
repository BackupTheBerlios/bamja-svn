017    package org.bamja.core;
018    
019    import java.util.List;
020    
021    import org.bamja.core.impl.BundleManager;
022    import org.bamja.core.interfaces.DependencyMetadata;
023    import org.bamja.core.interfaces.InstanceMetadata;
024    import org.osgi.framework.BundleActivator;
025    import org.osgi.framework.BundleContext;
026    
027    /**
028     * The GenericActivator is the default implementation of the BundleActivator
029     * interface. It starts the management of the bundle and at the end, it starts
030     * the terminate of it.
031     * <p>
032     * Der GenericActivator ist die Standardimplementierung vom Interface
033     * BundleActivator. Es startet das Managment des Bundles und startet am Ende
034     * desen Beendung.
035     * 
036     * @author Humberto Cervantes
037     * @author Jens Kutzsche
038     * @since Version: 0.2 (15.02.2006)
039     * @see BundleActivator
040     */
041    public abstract class GenericActivator implements BundleActivator {
042    
043        private BundleContext context = null;
044    
045        private BundleManager bundleManager = null;
046    
047        /**
048         * Called upon starting of the bundle from the framework. It's invoke
049         * startManage() of the BundleManager for starting the management of this
050         * bundle.
051         * <p>
052         * Wird vom Framework aufgerufen, wenn das Bundle gestartet wird. Es ruft
053         * startManage() vom BandleManager auf, um die Verwaltung des Bundles zu
054         * starten.
055         * 
056         * @param context The context of this bundle passed by the framework.
057         * @exception Exception Any exception thrown during starting of the
058         *                services.
059         */
060        public void start(BundleContext context) throws Exception {
061            this.context = context;
062            this.bundleManager = new BundleManager();
063    
064            this.bundleManager.startManage(context);
065        }
066    
067        /**
068         * Called upon stoping of the bundle from the framework. It's invoke
069         * stopManage() of the BundleManager for terminate all services.
070         * <p>
071         * Wird vom Framework aufgerufen, wenn das Bundle gestoppt wird. Es ruft
072         * stopManage() des BundleManager auf, um alle Dienste zu beenden.
073         * 
074         * @param context The context of this bundle passed by the framework
075         * @exception Exception Any exception thrown during terminate of the
076         *                services.
077         */
078        public void stop(BundleContext context) throws java.lang.Exception {
079            this.bundleManager.stopManage(context);
080        }
081    
082        /**
083         * Returns the BundleContext of this bundle.
084         * 
085         * @return The BundleContext of this bundle.
086         */
087        protected BundleContext getBundleContext() {
088            return this.context;
089        }
090    }
